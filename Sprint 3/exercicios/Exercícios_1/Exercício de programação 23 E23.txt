class Calculo:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def soma(self):
        return self.x + self.y
        
    def sub(self):
      return self.x - self.y
        
if __name__ == '__main__':
    calculo = Calculo(4,5)
    print("Somando:", calculo.soma())
    print("Subtraindo:", calculo.sub())
