class Aviao:
    def __init__(self, modelo, velocidade_maxima, capacidade, cor = "Cor azul"):
        self.modelo = modelo
        self.velocidade_maxima = velocidade_maxima
        self.capacidade = capacidade
        self.cor = cor
        
if __name__ == '__main__':
    aviao1 = Aviao(["BOIENG456", 1500, 400])

    aviao3 = Aviao(["Antonov An-2", 258, 12])
    print (f'modelo BOIENG456:{aviao1.modelo} velocidade máxima {aviao1.velocidade_maxima} km/h: capacidade para {aviao1.capacidade} passageiros: {aviao1.cor}')
    print (f'modelo BOIENG456:{aviao2.modelo} velocidade máxima {aviao2.velocidade_maxima} km/h: capacidade para {aviao2.capacidade} passageiros: {aviao2.cor}')
